This project's `platformio.ini` currently uses custom pre-built `platform_packages` that contain custom built ESP-IDF components. This prevents a situation when a default configuration ESP-IDF build is occasionally pulled into the build, and then the memory explodes and HFP doesn't work. It also dramatically speeds up the build process.

However, it does make those bits of code a bit of a black box. If you want to build the ESP-IDF components with the required customizations, then remove the `platform_packages` from `platformio.ini` and then also insert the following into `platformio.ini`:

```
custom_sdkconfig = 
    CONFIG_BT_SPP_ENABLED=y
    CONFIG_BT_HFP_ENABLE=y
    CONFIG_BT_HFP_CLIENT_ENABLE=y
    CONFIG_BT_HFP_AG_ENABLE=n
    CONFIG_BT_LOG_APPL_TRACE_LEVEL=2
    CONFIG_BT_LOG_RFCOMM_TRACE_LEVEL=2
    CONFIG_BT_LOG_SDP_TRACE_LEVEL=2
    CONFIG_BTDM_CTRL_BR_EDR_MAX_ACL_CONN=2
    CONFIG_BTDM_CTRL_BR_EDR_MAX_SYNC_CONN=1
    CONFIG_BTDM_CTRL_BR_EDR_MAX_SYNC_CONN_EFF=1
    CONFIG_BT_ACL_CONNECTIONS=4
    CONFIG_BT_HFP_WBS_ENABLE=n
    CONFIG_BT_HFP_AUDIO_DATA_PATH_HCI=y
    CONFIG_BT_HFP_AUDIO_DATA_PATH_PCM=n
    CONFIG_BTDM_CTRL_BR_EDR_SCO_DATA_PATH_HCI=y
    CONFIG_BTDM_CTRL_BR_EDR_SCO_DATA_PATH_PCM=n
    CONFIG_BT_GATTS_ENABLE=y
    CONFIG_BT_GATTC_ENABLE=y
    CONFIG_BT_BLE_BLUFI_ENABLE=n
    CONFIG_BT_BLE_SMP_ENABLE=y
    CONFIG_BLE_MESH=n
    CONFIG_BTDM_BLE_MESH_SCAN_DUPL_EN=n
    CONFIG_BLE_MESH_SCAN_DUPLICATE_EN=n
    CONFIG_BT_MULTI_CONNECTION_ENBALE=y
    CONFIG_BT_SMP_MAX_BONDS=15
    CONFIG_BT_ALARM_MAX_NUM=50
    CONFIG_ESP32_REV_MIN_3=y
    CONFIG_ESP32_REV_MIN=3
    CONFIG_ESP32_REV_MIN_FULL=300
    CONFIG_ESP_REV_MIN_FULL=300
    CONFIG_ESPTOOLPY_FLASHSIZE_4MB=n
    CONFIG_ESPTOOLPY_FLASHSIZE_16MB=y
    CONFIG_ESPTOOLPY_FLASHSIZE="16MB"
    CONFIG_ESP_WIFI_IRAM_OPT=n
    CONFIG_ESP_WIFI_RX_IRAM_OPT=n
    CONFIG_SPI_MASTER_ISR_IN_IRAM=n
    CONFIG_SPI_SLAVE_ISR_IN_IRAM=n
    CONFIG_SPIRAM_CACHE_WORKAROUND=n
    CONFIG_SPIRAM_CACHE_LIBCHAR_IN_IRAM=n
    CONFIG_SPIRAM_CACHE_LIBENV_IN_IRAM=n
    CONFIG_SPIRAM_CACHE_LIBFILE_IN_IRAM=n
    CONFIG_SPIRAM_CACHE_LIBIO_IN_IRAM=n
    CONFIG_SPIRAM_CACHE_LIBJMP_IN_IRAM=n
    CONFIG_SPIRAM_CACHE_LIBMATH_IN_IRAM=n
    CONFIG_SPIRAM_CACHE_LIBMEM_IN_IRAM=y
    CONFIG_SPIRAM_CACHE_LIBMISC_IN_IRAM=n
    CONFIG_SPIRAM_CACHE_LIBNUMPARSER_IN_IRAM=n
    CONFIG_SPIRAM_CACHE_LIBRAND_IN_IRAM=n
    CONFIG_SPIRAM_CACHE_LIBSTR_IN_IRAM=n
    CONFIG_SPIRAM_CACHE_LIBTIME_IN_IRAM=n
    CONFIG_BT_STACK_NO_LOG=n
    CONFIG_HAL_DEFAULT_ASSERTION_LEVEL=CONFIG_HAL_ASSERTION_DISABLE
    CONFIG_COMPILER_STACK_CHECK=y
    CONFIG_COMPILER_STACK_CHECK_MODE=STRONG
    CONFIG_COMPILER_STACK_CHECK_MODE_STRONG=y
    CONFIG_COMPILER_STACK_CANARY_MODE_ALL=y
    CONFIG_ESP_SYSTEM_MEMPROT_FEATURE=y
    CONFIG_ESP_SYSTEM_MEMPROT_FEATURE_LOCK=y
    CONFIG_HEAP_POISONING_LIGHT=y
    CONFIG_HEAP_TRACING_OFF=y
```

To explain what this does:

 * enables HFP for Bluetooth
 * shrinks the memory used for Bluetooth a bit
 * removes unused features from Bluetooth
 * removes or optimizes IRAM0 usage, otherwise it overflows
 * memory protection to prevent unsafe code execution
